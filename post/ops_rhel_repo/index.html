RHEL更换CentOS Yum

Jun 12, 2022 #yum #CentOS7 #epel-release #7.9.2009

v7.x

Red Hat Enterprise Linux7.x

Yum Repo

rpm -aq|grep yum|xargs rpm -e --nodeps
rpm -ivh http://mirrors.ustc.edu.cn/centos/7.9.2009/os/x86_64/Packages/wget-1.14-18.el7_6.1.x86_64.rpm

down:

wget http://mirrors.ustc.edu.cn/centos/7.9.2009/os/x86_64/Packages/yum-metadata-parser-1.1.4-10.el7.x86_64.rpm
wget http://mirrors.ustc.edu.cn/centos/7.9.2009/os/x86_64/Packages/yum-3.4.3-168.el7.centos.noarch.rpm
wget http://mirrors.ustc.edu.cn/centos/7.9.2009/os/x86_64/Packages/yum-plugin-fastestmirror-1.1.31-54.el7_8.noarch.rpm

install:

rpm -ivh yum-metadata-parser-1.1.4-10.el7.x86_64.rpm
rpm -ivh yum-3.4.3-168.el7.centos.noarch.rpm yum-plugin-fastestmirror-1.1.31-54.el7_8.noarch.rpm

repo file

/etc/yum.repos.d/CentOS-Base.repo CentOS-Base.repo - cento7

# CentOS-Base.repo
#
# The mirror system uses the connecting IP address of the client and the
# update status of each mirror to pick mirrors that are updated to and
# geographically close to the client.  You should use this for CentOS updates
# unless you are manually picking other mirrors.
#
# If the mirrorlist= does not work for you, as a fall back you can try the
# remarked out baseurl= line instead.
#
#
 
[base]
name=CentOS-7.9.2009 - Base
#mirrorlist=http://mirrorlist.centos.org/?release=7.9.2009&arch=x86_64&repo=os
baseurl=http://mirrors.ustc.edu.cn/centos/7.9.2009/os/x86_64/
gpgcheck=1
gpgkey=file:///etc/pki/rpm-gpg/RPM-GPG-KEY-CentOS-7
 
#released updates
[updates]
name=CentOS-7.9.2009 - Updates
# mirrorlist=http://mirrorlist.centos.org/?release=7.9.2009&arch=x86_64&repo=updates
baseurl=http://mirrors.ustc.edu.cn/centos/7.9.2009/updates/x86_64/
gpgcheck=1
gpgkey=file:///etc/pki/rpm-gpg/RPM-GPG-KEY-CentOS-7
 
#additional packages that may be useful
[extras]
name=CentOS-7.9.2009 - Extras
# mirrorlist=http://mirrorlist.centos.org/?release=7.9.2009&arch=x86_64&repo=extras
baseurl=http://mirrors.ustc.edu.cn/centos/7.9.2009/extras/x86_64/
gpgcheck=1
gpgkey=file:///etc/pki/rpm-gpg/RPM-GPG-KEY-CentOS-7
 
#additional packages that extend functionality of existing packages
[centosplus]
name=CentOS-7.9.2009 - Plus
# mirrorlist=http://mirrorlist.centos.org/?release=7.9.2009&arch=x86_64&repo=centosplus
baseurl=http://mirrors.ustc.edu.cn/centos/7.9.2009/centosplus/x86_64/
gpgcheck=1
enabled=0
gpgkey=file:///etc/pki/rpm-gpg/RPM-GPG-KEY-CentOS-7

install Extra Packages for Enterprise Linux (EPEL)

sudo rpm --import http://mirrors.ustc.edu.cn/centos/RPM-GPG-KEY-CentOS-7

sudo yum -y install epel-release
sudo yum clean all
sudo yum makecache
sudo yum repolist all
