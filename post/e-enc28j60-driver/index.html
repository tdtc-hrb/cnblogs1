uint32_t sysclk_get_peripheral_hz()
{
	/* CONFIG_SYSCLK_PRES is the register value for setting the expected */
	/* prescaler, not an immediate value. */
	return sysclk_get_main_hz() /
		((CONFIG_SYSCLK_PRES == SYSCLK_PRES_3) ? 3 :
			(1 << (CONFIG_SYSCLK_PRES >> PMC_MCKR_PRES_Pos)));
}

spi calc baudrate div

short spi_calc_baudrate_div(const uint32_t baudrate, uint32_t mck)
{
    int baud_div = div_ceil(mck, baudrate);

    /* The value of baud_div is from 1 to 255 in the SCBR field. */
    if (baud_div <= 0 || baud_div > 255) {
        return -1;
    }

    return baud_div;
}
div ceil
#define div_ceil(a, b)      (((a) + (b) - 1) / (b))

Set clock default state

void spi_set_clock_polarity(uint32_t ul_polarity)
{
    if (ul_polarity) {
        REG_SPI_CSR |= SPI_CSR_CPOL;
    } else {
        REG_SPI_CSR &= (~SPI_CSR_CPOL);
    }
}

Set Data Capture Phase

void spi_set_clock_phase(uint32_t ul_phase)
{
    if (ul_phase) {
        REG_SPI_CSR |= SPI_CSR_NCPHA;
    } else {
        REG_SPI_CSR &= (~SPI_CSR_NCPHA);
    }
}

void spi_put(uint16_t data)
{

	REG_SPI_TDR = SPI_TDR_TD(data);
}
