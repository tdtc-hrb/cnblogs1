support c++17

Configuration Properties > C/C++ > Command Line property page.

设置qt对象响应事件

首先，声明方法；然后，编写方法；最后，关联qt对象。

声明

注意： 声明是在slots下！

private slots:
	void helloQT();

方法

引用QMessageBox

#include <QtWidgets/QMessageBox>

完成后的方法

void QtHello::helloQT()
{
	QMessageBox msgBox;
	msgBox.setText("Hello Qt5!");
	msgBox.exec();
}

关联QT

在构造方法中使用connect对象与响应事件。 

connect(ui.btnHello, SIGNAL(clicked()), this, SLOT(helloQT()));

code

QtHello.h

#pragma once

#include <QtWidgets/QMainWindow>
#include "ui_QtHello.h"

class QtHello : public QMainWindow
{
    Q_OBJECT

public:
    QtHello(QWidget *parent = Q_NULLPTR);

private:
    Ui::QtHelloClass ui;
// add code
private slots:
    void helloQT();
};

QtHello.cpp

#include "QtHello.h"
#include <QtWidgets/QMessageBox>

QtHello::QtHello(QWidget *parent)
    : QMainWindow(parent)
{
    ui.setupUi(this);

	//
	connect(ui.btnHello, SIGNAL(clicked()), this, SLOT(helloQT()));
}


void QtHello::helloQT()
{
	QMessageBox msgBox;
	msgBox.setText("Hello Qt6.2!");
	msgBox.exec();
}
