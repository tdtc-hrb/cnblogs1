Flyweight Design Pattern Explained With Simple Example

  • flyweight
public abstract class Model {
	protected String m_name;
	protected int m_capacity;
	protected int m_speed;

	public Model(String name, int capacity, int speed) {
		this.m_name = name;
		this.m_capacity = capacity;
		this.m_speed = speed;
	}
	
	public void showDetails() {
		// TODO Auto-generated method stub
		System.out.println("Name: " + this.m_name + ", Capactiy: " 
		+ this.m_capacity + ", Speed: " + this.m_speed + " knots");
	}
}

  • Flyweight factory
public class FlyweightFactory {

	private Model model;
	
	public Model getModel(int type) {
		switch (type) {
		case 380:
			model = new Airbus380("Airbus380", 200, 800);
			break;

		case 787:
			model = new Boeing787("Boeing787", 600, 1000);
			break;
			
		case 797:
			model = new Boeing797("Boeing797", 1200, 1500);
			break;
		default:
			System.out.println("Unknown aeroplane type");
			model = null;
			break;
		}
		return model;
	}
}

public class Aeroplane {
	private Model m_model;
	private String m_mfgDate;
	private int m_id;
	
	public Aeroplane(Model model, String date, int id) {
		// TODO Auto-generated constructor stub
		this.m_model = model;
		this.m_mfgDate = date;
		this.m_id = id;
	}
	
	public void showAeroplane() {
		// TODO Auto-generated method stub
		m_model.showDetails();
		System.out.println("MfgDate : " + m_mfgDate + ", Serial No: "  + m_id);
	}
}
